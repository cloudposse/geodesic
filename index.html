#!/bin/bash # This is the geodesic bootstrap installer. # Just run: curl geodesic.sh | bash GEODESIC_SH=/usr/local/bin/geodesic curl -s --location http://geodesic.sh/dist/contrib/geodesic --output ${GEODESIC_SH} if [ $? -ne 0 ]; then echo "Failed to install to ${GEODESIC_SH}" exit 1 fi chmod 755 ${GEODESIC_SH} echo "Installed to ${GEODESIC_SH}"
